<script>
    document.getElementById("accordion1").querySelector(".description").style.display = "block"; // Set first item expanded by default
    function toggleAccordion(id) {
        var accordion = document.getElementById("accordion" + id);
        var description = accordion.querySelector(".description");
        var isExpanded = description.style.display === "block";
        var multiselect = document.getElementById("multiselect").checked;

        if (!multiselect) {
            // Collapse all first
            document.querySelectorAll(".description").forEach(function(desc) {
                desc.style.display = "none";
                desc.previousElementSibling.querySelector(".expand-icon").style.display = "inline";
                desc.previousElementSibling.querySelector(".collapse-icon").style.display = "none";
            });
        }

        // Toggle the clicked accordion
        if (isExpanded) {
            description.style.display = "none";
            accordion.querySelector(".expand-icon").style.display = "inline";
            accordion.querySelector(".collapse-icon").style.display = "none";
        } else {
            description.style.display = "block";
            accordion.querySelector(".expand-icon").style.display = "none";
            accordion.querySelector(".collapse-icon").style.display = "inline";
        }
    }
</script>
